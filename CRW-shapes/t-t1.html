f = plt.figure()
ax = f.add_subplot(1,2,1)
ax2 = f.add_subplot(1,2,2)
for b in beta:
    P=2*b*I/np.tan(theta)-0.8*b*(1-np.cos(theta)**2.5)
    iso = b*(theta/np.tan(theta)-1)
    ax.plot(np.degrees(theta),iso,"b--",label=r"isotropic, $\beta={}$".format(b))
    ax.plot(np.degrees(theta),P,"g-",label=r"proplyd, $\beta={}$".format(b))
    ax2.plot(np.degrees(theta),np.abs((P-iso)/iso),label=r"$\beta={}$".format(b))
ax.legend(loc="best",prop=dict(size="x-small"))
ax2.legend(loc="best")
ax.set_xlabel(r"$\theta$ (deg)")
ax.grid()
ax2.grid()
ax2.set_xlabel(r"$\theta$ (deg)")
ax.set_ylabel(r"$\theta_1\cot\theta_1-1$")
ax2.set_ylabel(r"$\epsilon$")
ax2.set_title(r"Relative difference for $\theta_1\cot\theta_1$")
ax.set_ylim(-0.01,0.001)
f.set_size_inches(10,5);
