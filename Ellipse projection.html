import numpy as np

def theta(t, a, b, x0):
    return np.degrees(np.arctan2(b*np.sin(t), a*np.cos(t) - x0))

def thetac(t, a, b, x0):
    return np.degrees(np.arctan2(b*np.sin(t), a*np.cos(t)))

def thetaprime(t, inc, a, b, x0):
    numerator = b*np.sqrt(np.sin(t)**2 - ((b/a)*np.tan(inc)*np.cos(t))**2)
    denominator = np.cos(inc)*(a*np.cos(t)*(1.0 + (b*np.tan(inc)/a)**2) - x0)
    return np.degrees(np.arctan2(numerator, denominator))

def g(inc, A, thetac):
    return 1 + A*(f(inc, thetac) - 1.0)/np.tan(thetac)**2

def f(inc, thetac):
    return np.sqrt(1 + np.tan(thetac)**2 * np.tan(inc)**2)

def Aprime(inc, A, thetac):
    return A/(np.cos(inc)**2 * f(inc, thetac) * g(inc, A, thetac))

A = 1.6

The dots are shown every \(15^\circ\) up to \(75^\circ\). There is a limiting value of \(A'\) for each value of \(\theta_ {\mathrm{c}}\), which is the limit as \(i \to 90^\circ\). This is 1 for circles \(\theta_ {\mathrm{c}} = 45^\circ\) and 2 for parabolas \(\theta_ {\mathrm{c}} = 0\).
