def yhyperbol(x, th_inf=45.0):
    tan_th = np.tan(np.radians(th_inf))
    return (np.sqrt(1.0 + x**2*tan_th**2) - 1.0)/tan_th**2

def yparabol(x):
    return 0.5*x**2

def ycircle(x):
    return 1.0 - np.sqrt(1.0 - x**2)
